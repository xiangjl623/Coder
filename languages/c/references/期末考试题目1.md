五、编程题参考代码
1. 阶乘和
```c
#include <stdio.h>
int main() {
    int n, i, j;
    long long fact = 1, sum = 0;
    scanf("%d", &n);
    for (i = 1; i <= n; i++) {
        fact = 1;
        for (j = 1; j <= i; j++)
            fact *= j;
        sum += fact;
    }
    printf("%lld", sum);
    return 0;
}
```

2. 结构体学生
```c
#include <stdio.h>
struct Student {
    int id;
    char name[20];
    float score;
};
int main() {
    struct Student s[3], max;
    int i;
    for (i = 0; i < 3; i++)
        scanf("%d%s%f", &s[i].id, s[i].name, &s[i].score);
    max = s[0];
    for (i = 1; i < 3; i++)
        if (s[i].score > max.score)
            max = s[i];
    printf("%d %s %.2f", max.id, max.name, max.score);
    return 0;
}
```

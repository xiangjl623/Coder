## 三、程序分析（3题，每题5分，共15分）

写出下列程序的运行结果。

**1.**
```c
#include <stdio.h>
main() {
    int a = 1, b = 10;
    do {
        b = a;
        a++;
    } while (b-- < 0);
    printf("a=%d,b=%d\n", a, b);
}
```

**2.**
```c
int sum(int a) {
    auto int c = 0;
    static int b = 3;
    c += 1;
    b += 2;
    return (a + b + c);
}
void main() {
    int i;
    int a = 2;
    for (i = 0; i < 5; i++) {
        printf("%d,", sum(a));
    }
}
```

**3.**
```c
#include <stdio.h>
void main() {
    int a = 10, b = 5, c = 5, d = 5;
    int i = 0, j = 0, k = 0;
    for (; a > b; ++b) i++;
    while (a > ++c) j++;
    do k++; while (a > d++);
    printf("%d,%d,%d\n", i, j, k);
}
```

    int i, j;
    char temp;
    for (i = 0, j = strlen(str) - 1; i < j; i++, j--) {
        temp = str[i];
        str[i] = str[j];
        str[j] = temp;
    }
}
